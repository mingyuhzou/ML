def select_threshold(pval,yval):
    best_epsilon=0
    best_f1=0
    f1=0
    # 划分100个阈值
    step=(pval.max()-pval.min())/1000
    for epsilon in np.arange(pval.min(),pval.max(),step):
        # 标签1表示true
        preds=pval<epsilon #  小于阈值表示异常值
        # 用于计算查全率和查准率
        # np.logical_ans求解两个响亮的逻辑与
        tp=np.sum(np.logical_and(preds==1,yval==1)).astype(np.float64)
        fp=np.sum(np.logical_and(preds==1,yval==0)).astype(np.float64)
        fn=np.sum(np.logical_and(preds==0,yval==1)).astype(np.float64)
        precision=tp/(tp+fp)
        recall=tp/(tp+fn)
        f1=2*precision*recall/(precision+recall)
        # 找出最好的f1值
        if f1>best_f1:
            best_f1=f1
            best_epsilon=epsilon
    return best_epsilon,best_f1
