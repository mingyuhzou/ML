def regularized_cost(theta,X,y,learning_rate):
    theta=theta.reshape(-1,1)
    inner=np.power(X@theta-y,2)
    return np.sum(inner)/(len(X)*2)+learning_rate/(2*len(X))*np.sum(np.power(theta[:,1:],2))

def gradient(theta,X,y):
    theta=theta.reshape(-1,1)
    grad=np.zeros(theta.shape[0])
    error=X@theta-y
    for j in range(theta.shape[0]):
        term=error*X[:,j].reshape(-1,1)
        grad[j]=np.sum(term)/len(X)
    return grad

def regularized_gradient(theta,X,y,learning_rate):
    theta=theta.reshape(-1,1)
    grad=np.zeros(theta.shape[0])
    error=X@theta-y
    for j in range(theta.shape[0]):
        term=error*X[:,j].reshape(-1,1)
        if not j:
            grad[j]=np.sum(term)/len(X)
        else:
            grad[j]=np.sum(term)/len(X)+learning_rate/len(X)*theta[j]
    return grad

def linear_regression_np(X,y,l=1):
    theta=np.ones(X.shape[1])
    res=opt.minimize(fun=regularized_cost,x0=theta,args=(X,y,l),method='TNC',jac=regularized_gradient,options={'disp':True})

def polynomial_feature(X,degree):
    X_poly=np.ones((len(X),degree+1))
    for i in range(1,degree+1):
        term=np.power(X,i)
        # 
        term=(term-term.mean())/term.std()
        X_poly[:,i]=term.ravel()
    return X_poly
