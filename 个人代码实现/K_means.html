通过比较数据点到给定的聚类中心的距离，取其中最小的

def find_closest_centroids(X,centroids):
    m=X.shape[0]
    k=centroids.shape[0]
    idx=np.zeros(m)
    for i in range(m):
        min_dist=inf
        for j in range(k):
            dist=np.sum((X[i,:]-centroids[j,:])**2)
            if dist<min_dist:
                min_dist=dist
                idx[i]=j
    return idx

def compute_centre(X,idx,k):
    m,n=X.shape
    res=np.zeros((k,n))
    for i in range(k):
        pos=np.where(idx==i)[0] # 注意返回的是一个元组
        # 对同属一类的所有数据点按列求和，得到新的聚类中心点
        res[i,:]=(np.sum(X[pos,:],axis=0)/len(pos)).ravel()
    return res

def run_k_means(X,init_centre,max_iters):
    m,n=X.shape
    k=init_centre.shape[0]
    idx=np.zeros(m)
    centre=init_centre
    for _ in range(max_iters):
        idx=find_closest_centroids(X,centre) # 求解出每个数据点的聚类中心
        centre=compute_centre(X,idx,k) # 求解新的聚类中心
    return idx,centre

def init_centroids(X,k):
    m,n=X.shape
    centroids=np.zeros((k,n))
    pos=np.random.randint(0,m,k)
    for i in range(k):
        centroids[i,:]=X[pos[i],:]
    return centroids
